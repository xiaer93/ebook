                //工厂模式产生单例对象，定时触发器！
                var timer=(function () {
                    var obj={};
                    var state=false;

                    function _go() {
                        obj.time=setTimeout(function () {
                            obj.callback();
                            _go();
                        },obj.duration)
                    }

                    //定制定时器
                    obj.start=function (callback,duration) {
                        obj.callback=callback;
                        obj.duration=duration;
                        _go();
                        state=true;
                    };
                    //清除定时器
                    obj.stop=function () {
                        if(obj.time){
                            clearTimeout(obj.time);
                            state=false;
                        }
                    };
                    //接着启动
                    obj.restart=function () {
                        //如果已经注册过事件，则继续执行！
                        if(obj.callback && state===false){
                            _go();
                            state=true;
                        }
                    };

                    return obj;
                })();

                var slideBox=document.querySelector(".m-weibo-slide .picture");
                var dotBoxa=document.querySelectorAll(".m-weibo-slide .dot-box a");

                /**
                 * 根据index切换幻灯片！
                 * @param index
                 */
                function slide(index) {
                    slideBox.style.transitionDuration="0.3s";
                    slideBox.style.transform="translate3d(-"+660*index+"px, 0px, 0px)";
                    if(index===6){
                        setTimeout(function () {
                            slideBox.style.transitionDuration="0s";
                            slideBox.style.transform="translate3d(-"+660+"px, 0px, 0px)";
                        },350);
                    }
                }

                /**
                 * 根据index切换按钮颜色！
                 * @param index
                 */
                function dotColor(index) {
                    dotBoxa.forEach(function (t) {
                        if(t.classList.contains("active")){
                            t.classList.toggle("active");
                        }
                        if(t.dataset.index===(index+'')){
                            t.classList.toggle("active");
                        }
                    })
                }

                //记录当前幻灯片索引
                var w_index=1;
                /**
                 * 微博幻灯片自动播放功能！
                 */
                function weiboAuto() {
                    w_index+=1;
                    slide(w_index);

                    if(w_index===6){
                        w_index=1;
                    }
                    dotColor(w_index);
                }
                /*
                * 给按钮绑定事件，实现通过按钮切换至指定图片页面！
                * */
                var dotBox=document.querySelector(".m-weibo-slide .dot-box");
                dotBox.addEventListener("click",function (e) {
                    //暂停自动播放
                    timer.stop();
                    //切换至选中的图片
                    var t=e.target;
                    var index=parseInt(t.dataset.index);
                    slide(index);
                    dotColor(index);
                    w_index=index;
                    //延时重新开始自动播放
                    //清除定时器，避免重复启动！

                    setTimeout(function () {
                        timer.restart();
                    },3000)
                },false);

                timer.start(weiboAuto,3500);
